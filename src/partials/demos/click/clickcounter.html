<div class="M_PreviewArea" data-demo="click-counter">
  <div class="demo-counter-number">0</div>
</div>
      

.demo-counter-number {
  font-size: 96px;
  font-weight: 900;
  color: #fff;
  user-select: none;
}
 
.demo-counter-number.bounce {
  animation: click-counter-bounce 0.3s ease-out;
}
 
@keyframes click-counter-bounce {
  0%   { transform: scale(1); }
  40%  { transform: scale(1.3); }
  70%  { transform: scale(0.95); }
  100% { transform: scale(1); }
}
      

function initClickCounter(root) {
  let count = 0
  const display = root.querySelector('.demo-counter-number')
 
  root.addEventListener('click', () => {
    count++
    display.textContent = count
 
    // Перезапуск CSS-анимации:
    // убираем класс, ждём новый кадр, добавляем обратно
    display.classList.remove('bounce')
    requestAnimationFrame(() => {
      display.classList.add('bounce')
    })
  })
}
